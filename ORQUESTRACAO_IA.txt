PASSO 5: ANALISAR RESULTADOS
----------------------------
Para análise de tendência C1/C2:

   import pyodbc
   
   DIV_C1 = {1, 3, 4}
   DIV_C2 = {15, 17, 18}
   
   conn_str = 'DRIVER={ODBC Driver 17 for SQL Server};SERVER=localhost;DATABASE=Lotofacil;Trusted_Connection=yes;'
   
   with pyodbc.connect(conn_str) as conn:
       cursor = conn.cursor()
       cursor.execute('SELECT TOP 20 Concurso, N1,N2,N3,N4,N5,N6,N7,N8,N9,N10,N11,N12,N13,N14,N15 FROM Resultados_INT ORDER BY Concurso DESC')
       
       for row in cursor.fetchall():
           resultado = set(row[i] for i in range(1, 16))
           d1 = len(resultado & DIV_C1)
           d2 = len(resultado & DIV_C2)
           fav = 'C1' if d1 > d2 else ('C2' if d2 > d1 else '==')
           print(f"Concurso {row.Concurso}: D1={d1}/3 D2={d2}/3 -> {fav}")
